구문

if문

if 문은 서로 다른 조건을 기반으로 다양한 구문을 실행하는 데 사용된다.


if (수식){
    실행문;
}


var a=5;
var b=7;
if(a<b){
    document.write("a는 b보다 작다");
}

if...else문

수식이 참인 경우 참이 아닌 경우 모두 실행한다.


if (수식){
    실행문;
}else{
    실행문;
}


var a=5;
var b=7;
if(a<b){
    document.write("a는 b보다 작다");
}
else{
  document.write("a는 b와 같거나 크다.");
}

if...else if...else 문

다양한 조건을 기반으로 여러 코드 중 하나를 실행하는 데 사용된다.


if (수식){
    실행문;
}else if(수식){
    실행문;
}else{
    실행문; 
}


var a=5;
var b=7;
if(a<b){
    document.write("a는 b보다 작다");
}else if(a==b){
    document.write("a는 b와 같다");
}else{
    document.write("a는 b보다 크다.");
}

switch 문

여러 경로로 분기하는 작업을 수행하기 위해 if 문을 사용할 수도 있다. 모든 분기가 한 변수의 값에 따라 좌우된다면 switch 문을 사용해볼 수 있다.


switch (수식){
    case 값1:
        실행문;
        break;
    case 값2:
        실행문;
        break;
    case 값3:
        실행문;
        break;
    default:
        실행문;
        break;
}


var x = 3;
switch (x){
    case 1:
        document.write("값은 1입니다");
        break;
    case 2:
        document.write("값은 2입니다");
        break;
    case 3:
        document.write("값은 3입니다");
        break;
    default:
        document.write("값은 1,2,3이 아닙니다.");
        break;
}

while 문

반복적인 동작을 수행할 때 사용된다.


while (수식){
    실행문;
}


var count = 1;
while (count < 5){
    document.write("count = " + count);
    count++;
}

do ... while 문

반복적인 동작을 수행하는 데 사용된다. do…while 문은 반복문 수식이 반복문의 맨 위가 아니라 맨 밑에서 검사된다는 점을 제외하면 while 문과 비슷하다. 이는 반복문의 본문이 늘 적어도 한 번은 실행된다는 것을 의미한다.


do{
    실행문;
} while (수식);


var count = 1;
do {
    document.write("count = " + count);
    count++;
} while(count < 5);

for 문

반복적인 동작을 수행하는 데 사용되며, 초기화식, 검사식, 증가식으로 세 부분으로 구성된다.


for (초기화식, 검사식, 증가식){
    실행문;
}


for (var count = 1; count < 5; count++){
    document.write("count = " + count);
}

break 문

가장 안쪽에 위치한 반복문이나 switch 문이 즉시 종료되게 한다.


break;

for (var count = 1; count < 10; count++){
    document.write("count = " + count);
    if (count == 3){ 
        break;
    }
}

continue 문

continue 문은 다음 번 순회로 반복문을 넘어가게 한다. continue 문은 while이나 do...while, for 반복문의 본문에서만 사용할 수 있다. continue 문이 실행되면 해당 구문을 감싸고 있는 반복문의 순회가 중단되고 다음 순회가 시작된다.


continue;

for (var count = 1; count < 6; count++){
    if (count == 3) {
        continue;
    }
    document.write("count = " + count);
}

return 문

return 문은 함수에서 반환되는 값을 지정하는 데 사용되는데, 함수 본문 안에서만 나타날 수 있다. return 문이 실행되면 수식이 평가되고 해당 함수의 값으로 반환된다. return 문이 실행되면 함수 본문에 다른 구문이 남아 있더라도 해당 함수의 실행이 중단된다.


return 수식;
